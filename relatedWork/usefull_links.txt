Install
git clone https://bitbucket.org/icl/papi.git
cd papi
git pull https://bitbucket.org/icl/papi.git
cd src
./configure
make
make test //to verify inatallation
or for ubuntu
papi install
sudo apt-get install papi-tools

    ------------------------------
