\section{Motivation and Background}

\begin{frame}{Interpreters}
    \begin{itemize}
        \item{Interpreters provide ease of implementation, and
            portability, at the expense of speed}
        \item{Easier to develop, maintain and port applications in new
            architectures}
        \item{Core of interpreter is an infinite loop that reads the
            next bytecode, decodes it and perform the appropriate
            action}
    \end{itemize}
\end{frame}

\begin{frame}{Problem state}
    \begin{itemize}
        \item{Overheads due to indirect branches}

        \item{We study the properties of current interpreters and
            state-of-the-art branch prediction on bare metal
            architectures}
    \end{itemize}
\end{frame}
